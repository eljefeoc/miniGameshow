**D. Create `showPasswordResetForm()` function:**

This function replaces the auth form content with a "set new password" form:

```js
function showPasswordResetForm(){
  const authEl = document.getElementById('ov-auth');
  if(!authEl) return;
  // Make sure auth panel is visible
  authEl.style.display = 'block';
  authEl.dataset.fpOpen = '1';

  // Hide all existing fields
  ['auth-email','auth-pass','auth-display-name','auth-age-label','auth-age-sub','auth-forgot-pw'].forEach(id=>{
    const el = document.getElementById(id);
    if(el) el.style.display = 'none';
  });

  const row = authEl.querySelector('.fp-auth-row');
  const msgEl = document.getElementById('auth-status');
  if(msgEl){ msgEl.textContent = 'Set your new password below.'; msgEl.style.color = '#7dffb0'; }

  // Show new password field and submit button
  const passEl = document.getElementById('auth-pass');
  if(passEl){
    passEl.style.display = '';
    passEl.value = '';
    passEl.placeholder = 'New password (8+ characters)';
    passEl.autocomplete = 'new-password';
  }

  if(row){
    row.innerHTML = '<button type="button" id="auth-set-new-pw" class="fp-btn-play btn-play" style="padding:12px 16px;min-height:44px;width:100%">Save new password</button>';
  }

  // Ensure overlay is visible
  const overlay = document.getElementById('overlay');
  if(overlay) overlay.style.display = 'flex';
}
```
